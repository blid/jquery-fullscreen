Fullscreen is inactive

Click me to toggle fullscreen!

Code used in this example: 

$('.toggle').click(function() {
    $('.fullscreen-container').fullscreen().toggle();
};

$(document).on('fullScreenChange', function(e, data) {
    var state = data.isActive ? 'active' : 'inactive';
    $('.state').removeClass('active inactive').addClass(state).text(state);
});
